import pandas as pd
from docx import Document

doc = Document('input1.docx')

table = doc.tables[0]
start_row = 2 # row that should become the column names
data = [[cell.text for cell in row.cells] for i, row in enumerate(table.rows) 
        if i >= start_row]
df = pd.DataFrame(data[1:], columns=data[0])

print(df)
